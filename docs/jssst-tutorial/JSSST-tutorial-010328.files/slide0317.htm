Meta-lock performance
! getMetaLock:
or   %i0, 3, %l0                ! %l0 = my busy value
add  %i1, 4, %l1                ! %l1 = lock word address
swap [%l1], %l0                 ! swap out busy value
and  %l0, 3, %l2                ! %l2 = meta-lock state
cmp  %l2, 3                     ! is lock state busy?
beq  slowGetMetaLock            ! branch if yes
sub  %l0, %l2, %l0              ! high 30 bits of lock word
! releaseMetaLock:
or   %i0, 3, %l0                ! %l0 = my busy value
cas  [%l1], %l0, %l2            ! if [%l1] == %l0 then swap [%l0],%l2
cmp  %l0, %l2                   ! did we do the swap?
bne  slowReleaseMetaLock        ! branch if no
